<script is:inline>
{`
(function() {
    function initCopiarColor() {
        const items = document.querySelectorAll('.color-item');
        items.forEach((item) => {
            item.addEventListener('click', () => {
                const colorCode = item.dataset.color;
                navigator.clipboard.writeText(colorCode);
                item.classList.add('copied');
                setTimeout(() => item.classList.remove('copied'), 800);
            });
        });
    }
    
    if (document.readyState === 'loading') {
        document.addEventListener('DOMContentLoaded', initCopiarColor);
    } else {
        initCopiarColor();
    }
    
    document.addEventListener('astro:page-load', initCopiarColor);
})();
`}
</script>
